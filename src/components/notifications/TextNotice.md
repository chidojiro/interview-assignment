```jsx
import { TextNotice } from '@ffw/randstad-shared-components';
import React, { useState } from 'react';

const showTextNotice = () => {
  const [textNotice, setTextNotice] = useState(true);
  const textNoticeCloseHandler = () => {
      setTextNotice(false);
  };
  return (
    <>
      {textNotice && (
        <TextNotice background="primary" onClose={textNoticeCloseHandler}>It looks like you are visiting from the United Kingdom.
          <a href="#">Take me to Randstad United Kindom </a>
        </TextNotice>
      )}
    </>
  );
};

showTextNotice();
```
